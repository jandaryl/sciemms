  mixins: [form],
  data () {
    return {
      config: {

        time_24hr: true,
        enableTime: true

      modelName: 'post',

      listPath: '/posts',
      tags: [],
      model: {
        title: null,
        summary: null,
        body: null,
        tags: [],
        featured_image: null,

        unpublished_at: null,

        promoted: false,
        meta: {
          title: null,
          description: null
        },
        has_featured_image: false
      }
    }
  },
  methods: {
    async getTags (search) {
      let {data} = await axios.get(this.$app.route('admin.tags.search'), {
        params: {
          q: search
        }
      })
 
      this.tags = data.items
    },
    deleteFeaturedImage () {
      this.$refs.featuredImageInput.reset()
      this.model.thumbnail_image_path = null
      this.model.has_featured_image = false
    }
  }
}
