<script>
import axios from 'axios'
import form from '../mixins/form'
 
export default {
  name: 'UserForm',
  mixins: [form],
  data () {
    return {
      roles: [],
      modelName: 'user',
      resourceRoute: 'users',
      listPath: '/users',
      model: {
        name: null,
        email: null,
        active: true,
        password: null,
        confirm_password: null,
        roles: []
      }
    }
  },
  async created () {
    this.fetchData()
    let {data} = await axios.get(this.$app.route(`admin.users.get_roles`))
    this.roles = data
  },
  methods: {
    onModelChanged () {
      if (this.model.roles) {
        this.model.roles = this.model.roles.map((item) => {
          return item.id
        })
      }
    }
  }
}
</script>
